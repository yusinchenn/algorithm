1. 演算法簡介

BFS (廣度優先搜尋)

佇列 (Queue) 地毯式搜索

像水波一樣層層擴散。它保證在無權重圖形中找到的絕對是「步數最少」的路徑。適合用於計算移動範圍或簡單迷宮尋路。

DFS (深度優先搜尋)

堆疊 (Stack) 嘗試錯誤

一條路走到黑，撞牆才回頭。路徑通常不是最短的，且形狀曲折。常用於生成迷宮、檢查地圖連通性或解決數獨問題。

Dijkstra (戴克斯特拉)

優先佇列 貪婪策略

考慮「成本」的 BFS。適合用於地形複雜（如沼澤移動慢、平地移動快）的地圖。它總是優先處理目前累計代價最小的節點。

2. 核心邏輯實作

// 使用 Queue (FIFO) 進行層層擴散
function bfs(startNode, endNode) {
    let queue = [startNode];
    let visited = new Set();
    visited.add(startNode);

    while (queue.length > 0) {
        let current = queue.shift(); // 取出隊首

        if (current === endNode) return reconstructPath(current);

        for (let neighbor of getNeighbors(current)) {
            if (!visited.has(neighbor)) {
                visited.add(neighbor);
                neighbor.parent = current; // 紀錄來源以便回溯路徑
                queue.push(neighbor);
            }
        }
    }
}
// 使用 Stack (LIFO) 進行深度探索
function dfs(startNode, endNode) {
    let stack = [startNode];
    let visited = new Set();
    visited.add(startNode);

    while (stack.length > 0) {
        let current = stack.pop(); // 取出堆疊頂部

        if (current === endNode) return reconstructPath(current);

        for (let neighbor of getNeighbors(current)) {
            if (!visited.has(neighbor)) {
                visited.add(neighbor);
                neighbor.parent = current;
                stack.push(neighbor);
            }
        }
    }
}
// 使用 Priority Queue (依成本排序)
function dijkstra(startNode, endNode) {
    // 初始化：起點距離為0，其他為無限大
    nodes.forEach(n => n.distance = Infinity);
    startNode.distance = 0;
    
    let pq = [startNode]; // 簡單模擬 Priority Queue

    while (pq.length > 0) {
        // 總是取出目前成本最小的節點
        pq.sort((a, b) => a.distance - b.distance);
        let current = pq.shift();

        if (current === endNode) return reconstructPath(current);

        for (let neighbor of getNeighbors(current)) {
            let newDist = current.distance + neighbor.weight;
            if (newDist < neighbor.distance) {
                neighbor.distance = newDist;
                neighbor.parent = current;
                pq.push(neighbor);
            }
        }
    }
}
